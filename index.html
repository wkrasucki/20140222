 double miejsceZerowe(double (*funkcja)(double), double a, double b, double epsilon){
if (a>=b) return NAN;
double ya, yb, x, y;

 ya=funkcja(a);
 yb=funkcja(b);
 if (ya*yb>=0) return NAN;
 do {x=(a+b)/2; y=funkcja(x);
 if (y==0) return x;
 if (ya*y<0){ b=x; yb=y;}
